from machine import Pin, PWM, time_pulse_us
import time

# 1. Initialization
MAX_LIMIT = 50
red_count = 0
green_count = 0

# TCS3200 Pin Setup
s2 = Pin(14, Pin.OUT)
s3 = Pin(27, Pin.OUT)
out_pin = Pin(26, Pin.IN)

# Servo Pin Setup (GPIO 18)
servo = PWM(Pin(18), freq=50)

def set_servo_angle(angle):
    """Function to rotate the servo motor according to the angle"""
    duty = int(((angle / 180) * 2 + 0.5) / 20 * 1023)
    servo.duty(duty)

def get_color_data():
    """Takes 10 color readings and returns the average (Averaging)"""
    r_total = 0
    g_total = 0
    
    for _ in range(10):
        # Reading Red color
        s2.value(0)
        s3.value(0)
        r_total += time_pulse_us(out_pin, 0)
        
        # Reading Green color
        s2.value(1)
        s3.value(1)
        g_total += time_pulse_us(out_pin, 0)
        
        time.sleep(0.01)
        
    return (r_total // 10), (g_total // 10)

# Set servo to middle position (90°) at start
set_servo_angle(90)
print("System Active. Sorting can begin...")

# 2. Main Loop
while True:
    # SENSING PHASE
    r_avg, g_avg = get_color_data()
    
    # Check if an object is present (If value > 500, assumes no object)
    if r_avg > 500 and g_avg > 500:
        continue 
        
    # DECISION PHASE
    target = "UNKNOWN"
    if r_avg < g_avg:
        target = "RED"
    elif g_avg < r_avg:
        target = "GREEN"
        
    # CAPACITY CHECK PHASE
    if target == "RED" and red_count >= MAX_LIMIT:
        print("WARNING: Red bin is full!")
        break # Stops the robot
        
    if target == "GREEN" and green_count >= MAX_LIMIT:
        print("WARNING: Green bin is full!")
        break

    # ACTION PHASE
    if target == "RED":
        print(f"Red pepper detected. Value: {r_avg}")
        set_servo_angle(45) # Direct to red bin
        red_count += 1
        time.sleep(2) # Give time for the pepper to fall
        
    elif target == "GREEN":
        print(f"Green pepper detected. Value: {g_avg}")
        set_servo_angle(135) # Direct to green bin
        green_count += 1
        time.sleep(2)

    # Return to neutral position
    set_servo_angle(90)
    print(f"Status: Red {red_count}, Green {green_count}")
    time.sleep(1)

MAX_CAPACITY = 50  # Set your bin limit here

# Pin Setup for TCS3200
s2 = Pin(14, Pin.OUT)
s3 = Pin(27, Pin.OUT)
out_pin = Pin(26, Pin.IN)

# Pin Setup for Servo (GPIO 18)
servo = PWM(Pin(18), freq=50)

def set_servo_angle(angle):
    # Converts 0-180 degrees to duty cycle (approx 40 to 115 for ESP32)
    duty = int(((angle / 180) * 2 + 0.5) / 20 * 1023)
    servo.duty(duty)

def get_stable_color():
    # Takes 5 readings and averages them to prevent "flicker" errors
    r_total, g_total = 0, 0
    for _ in range(5):
        s2.value(0); s3.value(0)
        r_total += time_pulse_us(out_pin, 0)
        s2.value(1); s3.value(1)
        g_total += time_pulse_us(out_pin, 0)
        time.sleep(0.02)
    return r_total // 5, g_total // 5

# Start-up: Move servo to neutral (center) position
set_servo_angle(90)
print("Robot Initialized. Ready to harvest Scotch Bonnets!")

while True:
    r, g = get_stable_color()
    
    # Logic: Only act if something is actually in front of the sensor (Low pulse)
    if r < 300 or g < 300:
        if r < g:
            color = "RED"
            if red_bin_count < MAX_CAPACITY:
                set_servo_angle(45) # Move arm to Red Bin
                red_bin_count += 1
                print(f"Ripe Red sorted! Red Bin: {red_bin_count}/{MAX_CAPACITY}")
            else:
                print("ALERT: Red Bin is FULL!")
        else:
            color = "GREEN"
            if green_bin_count < MAX_CAPACITY:
                set_servo_angle(135) # Move arm to Green Bin
                green_bin_count += 1
                print(f"Unripe Green sorted! Green Bin: {green_bin_count}/{MAX_CAPACITY}")
            else:
                print("ALERT: Green Bin is FULL!")
        
        time.sleep(2)        # Wait for pepper to slide down
        set_servo_angle(90)  # Return to neutral
    
    time.sleep(0.5) # Check every half-second

from machine import Pin, PWM, UART, time_pulse_us
import time

# 1. Setup Pins
TRIG = Pin(5, Pin.OUT)
ECHO = Pin(18, Pin.IN)
SERVO_PIN = Pin(19)
ALERT_LED = Pin(2, Pin.OUT)

# UART connection to Raspberry Pi
uart = UART(0, baudrate=115200)

# Servo Setup
servo = PWM(SERVO_PIN, freq=50)

# 2. Configuration
DISTANCE_THRESHOLD = 5  # Alert if chillies are closer than 5cm
red_count = 0
green_count = 0

def get_distance():
    """Measures distance using the Ultrasonic Sensor"""
    TRIG.value(0)
    time.sleep_us(2)
    TRIG.value(1)
    time.sleep_us(10)
    TRIG.value(0)
    
    # Measure the duration of the echo pulse
    duration = time_pulse_us(ECHO, 1, 30000) # 30ms timeout
    distance = (duration * 0.0343) / 2
    return distance

def set_servo_angle(angle):
    duty = int(((angle / 180) * 2 + 0.5) / 20 * 1023)
    servo.duty(duty)

def trigger_full_alert():
    """Stops the robot and blinks the alert LED"""
    print("!!! BIN IS PHYSICALLY FULL !!!")
    while True:
        ALERT_LED.value(1)
        time.sleep(0.2)
        ALERT_LED.value(0)
        time.sleep(0.2)

# Start in neutral
set_servo_angle(90)

while True:
    # Continuously check the physical bin level
    current_dist = get_distance()
    
    if current_dist > 0 and current_dist < DISTANCE_THRESHOLD:
        trigger_full_alert()

    # Wait for commands from Raspberry Pi YOLO
    if uart.any():
        command = uart.readline().decode('utf-8').strip().upper()
        
        if command == "RED":
            set_servo_angle(45)
            red_count += 1
            time.sleep(2)
            set_servo_angle(90)
            print(f"Sorted Red. Total: {red_count}")

        elif command == "GREEN":
            set_servo_angle(135)
            green_count += 1
            time.sleep(2)
            set_servo_angle(90)
            print(f"Sorted Green. Total: {green_count}")

    time.sleep(0.1)

from machine import Pin, PWM, UART, time_pulse_us
import time

# --- 1. PIN CONFIGURATION ---
# Arrival Sensor (at the sorting gate)
TRIG_GATE = Pin(5, Pin.OUT)
ECHO_GATE = Pin(18, Pin.IN)

# Bin Monitoring Sensors (Capacity)
# Red Bin
TRIG_RED = Pin(13, Pin.OUT)
ECHO_RED = Pin(12, Pin.IN)
# Green Bin
TRIG_GREEN = Pin(14, Pin.OUT)
ECHO_GREEN = Pin(27, Pin.IN)

# Actuators
SERVO = PWM(Pin(19), freq=50)
ALERT_LED = Pin(2, Pin.OUT)

# Communication (to Raspberry Pi)
uart = UART(0, baudrate=115200)

# --- 2. SETTINGS ---
DIST_THRESHOLD = 5    # Bin is full if pepper is closer than 5cm
GATE_THRESHOLD = 8    # Pepper arrived if closer than 8cm
NEUTRAL_ANGLE = 45    # Waiting position
RED_ANGLE = 90        # Red bin position
GREEN_ANGLE = 0       # Green bin position

def get_distance(trig, echo):
    """Measures distance in cm using ultrasonic sensor"""
    trig.value(0)
    time.sleep_us(2)
    trig.value(1)
    time.sleep_us(10)
    trig.value(0)
    duration = time_pulse_us(echo, 1, 30000)
    if duration < 0: return 999 # Timeout
    return (duration * 0.0343) / 2

def set_servo(angle):
    """Moves servo to specific degree"""
    duty = int(((angle / 180) * 2 + 0.5) / 20 * 1023)
    SERVO.duty(duty)

def check_capacities():
    """Check if any bin is physically full before sorting"""
    if 0 < get_distance(TRIG_RED, ECHO_RED) < DIST_THRESHOLD:
        trigger_alert("RED BIN FULL")
    if 0 < get_distance(TRIG_GREEN, ECHO_GREEN) < DIST_THRESHOLD:
        trigger_alert("GREEN BIN FULL")

def trigger_alert(message):
    """Halts the system and blinks LED"""
    print(f"!!! ALERT: {message} !!!")
    while True:
        ALERT_LED.value(1)
        time.sleep(0.2)
        ALERT_LED.value(0)
        time.sleep(0.2)

# --- 3. MAIN PROCESS ---
print("System Initialized. Waiting for peppers...")
set_servo(NEUTRAL_ANGLE)

while True:
    # A. Constant Capacity Monitoring
    check_capacities()

    # B. Object Arrival Detection
    dist_at_gate = get_distance(TRIG_GATE, ECHO_GATE)
    
    if 0 < dist_at_gate < GATE_THRESHOLD:
        print("Pepper detected at gate. Triggering YOLO...")
        uart.write(b"CHECK_COLOR\n") # Send signal to Raspberry Pi
        
        # C. Wait for Pi Decision
        start_wait = time.time()
        decision_made = False
        
        while (time.time() - start_wait) < 5: # 5 second timeout
            if uart.any():
                command = uart.readline().decode('utf-8').strip().upper()
                
                if command == "RED":
                    print("Sorting to RED box")
                    set_servo(RED_ANGLE)
                    decision_made = True
                elif command == "GREEN":
                    print("Sorting to GREEN box")
                    set_servo(GREEN_ANGLE)
                    decision_made = True
                
                if decision_made:
                    time.sleep(2) # Wait for pepper to fall
                    set_servo(NEUTRAL_ANGLE)
                    break
        
        if not decision_made:
            print("No response from Pi or unknown color.")

    time.sleep(0.1)
