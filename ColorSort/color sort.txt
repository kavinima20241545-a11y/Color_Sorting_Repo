from machine import Pin, PWM, time_pulse_us
import time

# 1. Initialization
MAX_LIMIT = 50
red_count = 0
green_count = 0

# TCS3200 Pin Setup
s2 = Pin(14, Pin.OUT)
s3 = Pin(27, Pin.OUT)
out_pin = Pin(26, Pin.IN)

# Servo Pin Setup (GPIO 18)
servo = PWM(Pin(18), freq=50)

def set_servo_angle(angle):
    """Function to rotate the servo motor according to the angle"""
    duty = int(((angle / 180) * 2 + 0.5) / 20 * 1023)
    servo.duty(duty)

def get_color_data():
    """Takes 10 color readings and returns the average (Averaging)"""
    r_total = 0
    g_total = 0
    
    for _ in range(10):
        # Reading Red color
        s2.value(0)
        s3.value(0)
        r_total += time_pulse_us(out_pin, 0)
        
        # Reading Green color
        s2.value(1)
        s3.value(1)
        g_total += time_pulse_us(out_pin, 0)
        
        time.sleep(0.01)
        
    return (r_total // 10), (g_total // 10)

# Set servo to middle position (90°) at start
set_servo_angle(90)
print("System Active. Sorting can begin...")

# 2. Main Loop
while True:
    # SENSING PHASE
    r_avg, g_avg = get_color_data()
    
    # Check if an object is present (If value > 500, assumes no object)
    if r_avg > 500 and g_avg > 500:
        continue 
        
    # DECISION PHASE
    target = "UNKNOWN"
    if r_avg < g_avg:
        target = "RED"
    elif g_avg < r_avg:
        target = "GREEN"
        
    # CAPACITY CHECK PHASE
    if target == "RED" and red_count >= MAX_LIMIT:
        print("WARNING: Red bin is full!")
        break # Stops the robot
        
    if target == "GREEN" and green_count >= MAX_LIMIT:
        print("WARNING: Green bin is full!")
        break

    # ACTION PHASE
    if target == "RED":
        print(f"Red pepper detected. Value: {r_avg}")
        set_servo_angle(45) # Direct to red bin
        red_count += 1
        time.sleep(2) # Give time for the pepper to fall
        
    elif target == "GREEN":
        print(f"Green pepper detected. Value: {g_avg}")
        set_servo_angle(135) # Direct to green bin
        green_count += 1
        time.sleep(2)

    # Return to neutral position
    set_servo_angle(90)
    print(f"Status: Red {red_count}, Green {green_count}")
    time.sleep(1)

from machine import Pin, PWM, time_pulse_us
import time

# --- CONFIGURATION ---
MAX_CAPACITY = 50  # Set your bin limit here
red_bin_count = 0
green_bin_count = 0

# Pin Setup for TCS3200
s2 = Pin(14, Pin.OUT)
s3 = Pin(27, Pin.OUT)
out_pin = Pin(26, Pin.IN)

# Pin Setup for Servo (GPIO 18)
servo = PWM(Pin(18), freq=50)

def set_servo_angle(angle):
    # Converts 0-180 degrees to duty cycle (approx 40 to 115 for ESP32)
    duty = int(((angle / 180) * 2 + 0.5) / 20 * 1023)
    servo.duty(duty)

def get_stable_color():
    # Takes 5 readings and averages them to prevent "flicker" errors
    r_total, g_total = 0, 0
    for _ in range(5):
        s2.value(0); s3.value(0)
        r_total += time_pulse_us(out_pin, 0)
        s2.value(1); s3.value(1)
        g_total += time_pulse_us(out_pin, 0)
        time.sleep(0.02)
    return r_total // 5, g_total // 5

# Start-up: Move servo to neutral (center) position
set_servo_angle(90)
print("Robot Initialized. Ready to harvest Scotch Bonnets!")

while True:
    r, g = get_stable_color()
    
    # Logic: Only act if something is actually in front of the sensor (Low pulse)
    if r < 300 or g < 300:
        if r < g:
            color = "RED"
            if red_bin_count < MAX_CAPACITY:
                set_servo_angle(45) # Move arm to Red Bin
                red_bin_count += 1
                print(f"Ripe Red sorted! Red Bin: {red_bin_count}/{MAX_CAPACITY}")
            else:
                print("ALERT: Red Bin is FULL!")
        else:
            color = "GREEN"
            if green_bin_count < MAX_CAPACITY:
                set_servo_angle(135) # Move arm to Green Bin
                green_bin_count += 1
                print(f"Unripe Green sorted! Green Bin: {green_bin_count}/{MAX_CAPACITY}")
            else:
                print("ALERT: Green Bin is FULL!")
        
        time.sleep(2)        # Wait for pepper to slide down
        set_servo_angle(90)  # Return to neutral
    
    time.sleep(0.5) # Check every half-second

4) Ai Camera Connected

from machine import Pin, PWM, UART
import time

# 1. Initialization
MAX_LIMIT = 50
red_count = 0
green_count = 0

# Setup UART (Serial) Communication
# Baud rate 115200 is standard for AI camera modules
uart = UART(2, baudrate=115200, tx=17, rx=16)

# Servo Pin Setup (GPIO 18)
servo = PWM(Pin(18), freq=50)

def set_servo_angle(angle):
    duty = int(((angle / 180) * 2 + 0.5) / 20 * 1023)
    servo.duty(duty)

set_servo_angle(90)
print("Waiting for YOLO data from AI Camera...")

while True:
    if uart.any():
        # Read the message from the AI camera
        data = uart.readline()
        message = str(data, 'utf-8').strip().upper()
        
        print(f"YOLO Detection: {message}")

        # DECISION & ACTION PHASE
        if "RED" in message:
            if red_count < MAX_LIMIT:
                set_servo_angle(45)
                red_count += 1
                print(f"Sorted Red. Total: {red_count}")
                time.sleep(2)
            else:
                print("Red Bin Full!")

        elif "GREEN" in message:
            if green_count < MAX_LIMIT:
                set_servo_angle(135)
                green_count += 1
                print(f"Sorted Green. Total: {green_count}")
                time.sleep(2)
            else:
                print("Green Bin Full!")

        # Reset Servo to center
        set_servo_angle(90)

    time.sleep(0.1)

from machine import Pin, PWM, UART
import sys

# 1. Setup
MAX_LIMIT = 50
red_count = 0
green_count = 0

# Using UART 0 (the one connected to the USB Port)
# On most ESP32 boards, this is the default Serial
uart = UART(0, baudrate=115200)

servo = PWM(Pin(18), freq=50)

def set_servo_angle(angle):
    duty = int(((angle / 180) * 2 + 0.5) / 20 * 1023)
    servo.duty(duty)

set_servo_angle(90) # Start at Neutral

while True:
    if uart.any():
        # Read the command from Raspberry Pi
        command = uart.readline().decode('utf-8').strip()
        
        if command == "RED":
            if red_count < MAX_LIMIT:
                set_servo_angle(45)
                red_count += 1
                print("ACK:RED_SORTED") # Tell Pi it's done
            else:
                print("ERR:RED_FULL")

        elif command == "GREEN":
            if green_count < MAX_LIMIT:
                set_servo_angle(135)
                green_count += 1
                print("ACK:GREEN_SORTED")
            else:
                print("ERR:GREEN_FULL")

        # Allow time for the pepper to drop, then reset
        import time
        time.sleep(2)
        set_servo_angle(90)
