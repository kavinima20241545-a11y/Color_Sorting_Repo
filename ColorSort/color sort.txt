from machine import Pin, PWM, time_pulse_us
import time

# 1. Initialization
MAX_LIMIT = 50
red_count = 0
green_count = 0

# TCS3200 Pin Setup
s2 = Pin(14, Pin.OUT)
s3 = Pin(27, Pin.OUT)
out_pin = Pin(26, Pin.IN)

# Servo Pin Setup (GPIO 18)
servo = PWM(Pin(18), freq=50)

def set_servo_angle(angle):
    """Function to rotate the servo motor according to the angle"""
    duty = int(((angle / 180) * 2 + 0.5) / 20 * 1023)
    servo.duty(duty)

def get_color_data():
    """Takes 10 color readings and returns the average (Averaging)"""
    r_total = 0
    g_total = 0
    
    for _ in range(10):
        # Reading Red color
        s2.value(0)
        s3.value(0)
        r_total += time_pulse_us(out_pin, 0)
        
        # Reading Green color
        s2.value(1)
        s3.value(1)
        g_total += time_pulse_us(out_pin, 0)
        
        time.sleep(0.01)
        
    return (r_total // 10), (g_total // 10)

# Set servo to middle position (90°) at start
set_servo_angle(90)
print("System Active. Sorting can begin...")

# 2. Main Loop
while True:
    # SENSING PHASE
    r_avg, g_avg = get_color_data()
    
    # Check if an object is present (If value > 500, assumes no object)
    if r_avg > 500 and g_avg > 500:
        continue 
        
    # DECISION PHASE
    target = "UNKNOWN"
    if r_avg < g_avg:
        target = "RED"
    elif g_avg < r_avg:
        target = "GREEN"
        
    # CAPACITY CHECK PHASE
    if target == "RED" and red_count >= MAX_LIMIT:
        print("WARNING: Red bin is full!")
        break # Stops the robot
        
    if target == "GREEN" and green_count >= MAX_LIMIT:
        print("WARNING: Green bin is full!")
        break

    # ACTION PHASE
    if target == "RED":
        print(f"Red pepper detected. Value: {r_avg}")
        set_servo_angle(45) # Direct to red bin
        red_count += 1
        time.sleep(2) # Give time for the pepper to fall
        
    elif target == "GREEN":
        print(f"Green pepper detected. Value: {g_avg}")
        set_servo_angle(135) # Direct to green bin
        green_count += 1
        time.sleep(2)

    # Return to neutral position
    set_servo_angle(90)
    print(f"Status: Red {red_count}, Green {green_count}")
    time.sleep(1)

from machine import Pin, PWM, time_pulse_us
import time

# --- CONFIGURATION ---
MAX_CAPACITY = 50  # Set your bin limit here
red_bin_count = 0
green_bin_count = 0

# Pin Setup for TCS3200
s2 = Pin(14, Pin.OUT)
s3 = Pin(27, Pin.OUT)
out_pin = Pin(26, Pin.IN)

# Pin Setup for Servo (GPIO 18)
servo = PWM(Pin(18), freq=50)

def set_servo_angle(angle):
    # Converts 0-180 degrees to duty cycle (approx 40 to 115 for ESP32)
    duty = int(((angle / 180) * 2 + 0.5) / 20 * 1023)
    servo.duty(duty)

def get_stable_color():
    # Takes 5 readings and averages them to prevent "flicker" errors
    r_total, g_total = 0, 0
    for _ in range(5):
        s2.value(0); s3.value(0)
        r_total += time_pulse_us(out_pin, 0)
        s2.value(1); s3.value(1)
        g_total += time_pulse_us(out_pin, 0)
        time.sleep(0.02)
    return r_total // 5, g_total // 5

# Start-up: Move servo to neutral (center) position
set_servo_angle(90)
print("Robot Initialized. Ready to harvest Scotch Bonnets!")

while True:
    r, g = get_stable_color()
    
    # Logic: Only act if something is actually in front of the sensor (Low pulse)
    if r < 300 or g < 300:
        if r < g:
            color = "RED"
            if red_bin_count < MAX_CAPACITY:
                set_servo_angle(45) # Move arm to Red Bin
                red_bin_count += 1
                print(f"Ripe Red sorted! Red Bin: {red_bin_count}/{MAX_CAPACITY}")
            else:
                print("ALERT: Red Bin is FULL!")
        else:
            color = "GREEN"
            if green_bin_count < MAX_CAPACITY:
                set_servo_angle(135) # Move arm to Green Bin
                green_bin_count += 1
                print(f"Unripe Green sorted! Green Bin: {green_bin_count}/{MAX_CAPACITY}")
            else:
                print("ALERT: Green Bin is FULL!")
        
        time.sleep(2)        # Wait for pepper to slide down
        set_servo_angle(90)  # Return to neutral
    
    time.sleep(0.5) # Check every half-second

import time

MAX_LIMIT = 50
counts = {"RED": 0, "GREEN": 0}
servo = PWM(Pin(18), freq=50)

def set_servo(angle):
    duty = int(((angle / 180) * 2 + 0.5) / 20 * 1023)
    servo.duty(duty)

# Initialize
set_servo(90)
# Use UART 0 which is the USB-Serial bridge
uart = UART(0, baudrate=115200)

while True:
    if uart.any():
        command = uart.readline().decode('utf-8').strip().upper()
        
        if command in counts:
            if counts[command] < MAX_LIMIT:
                # Move to bin: 45 for RED, 135 for GREEN
                angle = 45 if command == "RED" else 135
                set_servo(angle)
                
                counts[command] += 1
                print(f"ACK:{command}:{counts[command]}")
                
                time.sleep(2) # Wait for pepper to slide
                set_servo(90) # Return to neutral
            else:
                print(f"FULL:{command}")

from machine import Pin, PWM, UART
import time

# 1. Setup
MAX_LIMIT = 50
red_count = 0
green_count = 0

# Alert Pin (Using the built-in LED on GPIO 2, or connect a Buzzer)
alert_led = Pin(2, Pin.OUT)

# UART 0 connection to Raspberry Pi
uart = UART(0, baudrate=115200)

# Servo connected to GPIO 18
servo = PWM(Pin(18), freq=50)

def set_servo_angle(angle):
    duty = int(((angle / 180) * 2 + 0.5) / 20 * 1023)
    servo.duty(duty)

def trigger_alert():
    """Blinks the LED/Buzzer to signal a full bin"""
    while True:
        alert_led.value(1)
        time.sleep(0.5)
        alert_led.value(0)
        time.sleep(0.5)
        print("SYSTEM HALTED: PLEASE EMPTY THE BINS")

# Start in neutral
set_servo_angle(90)
alert_led.value(0)

while True:
    if uart.any():
        command = uart.readline().decode('utf-8').strip().upper()
        
        # --- RED CHILLI LOGIC ---
        if command == "RED":
            if red_count < MAX_LIMIT:
                set_servo_angle(45)
                red_count += 1
                time.sleep(2)
                set_servo_angle(90)
            else:
                print("ALERT: Red Bin Full!")
                trigger_alert() # This stops the whole code until reset

        # --- GREEN CHILLI LOGIC ---
        elif command == "GREEN":
            if green_count < MAX_LIMIT:
                set_servo_angle(135)
                green_count += 1
                time.sleep(2)
                set_servo_angle(90)
            else:
                print("ALERT: Green Bin Full!")
                trigger_alert() # This stops the whole code until reset

        print(f"Stats - Red: {red_count}/{MAX_LIMIT}, Green: {green_count}/{MAX_LIMIT}")
